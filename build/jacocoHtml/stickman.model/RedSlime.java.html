package stickman.model;

public class RedSlime extends Enemy{
    private AutoStrategy autoJump;
    private AutoStrategy autoMove;

    public RedSlime(double posX, double posY){
        super(posX, posY);
        this.velX = 2;
        this.velY = -2;
        this.mainPath = "slimeRa.png";

        this.width = 63;
        this.height = 45;

        this.Hp = 1;
        autoJump = new AutoJump();
        autoMove = new AutoXmove();
    }

    @Override
    void changYLocation() {
        autoJump.getInfo(this);
        posY = autoJump.getValue();
        autoJump.update(this);

    }

    @Override
    void changXLocation() {
        autoMove.getInfo(this);
        this.posX = autoMove.getValue();
        autoMove.update(this);
    }

    @Override
    void tick() {

        if( !isJumping && !isComingDown){
            changXLocation();
        }
        changYLocation();

        if( this.timer < 90){
            setXDirection(true);
        }else{
            setXDirection(false);
        }

        if (this.timer == 75 && !isComingDown && !isJumping){
            isJumping = true;
        }

        if (this.timer == 90 && isJumping){
            isJumping = false;
        }


        this.timer += 1;
        if( this.timer >= 180 ){
            timer = 0;
        }
    }
}
