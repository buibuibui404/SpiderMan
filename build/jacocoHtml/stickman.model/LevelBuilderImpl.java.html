package stickman.model;

import java.util.LinkedList;
import java.util.List;

public class LevelBuilderImpl implements LevelBuilder {
    private List<Entity> entities;
    private List<Platform> platforms;
    private List<Enemy> enemies;
    private List<BackgroundEntity> bg;
    private Hero hero;
    private double finishLine;

    public LevelBuilderImpl(){
        entities = new LinkedList<>();
        platforms = new LinkedList<>();
        enemies = new LinkedList<>();
        bg = new LinkedList<>();
    }

    @Override
    public void setHero(Hero h) {
        hero = h;
        addEntity(h);
    }

    @Override
    public void addEntity(Entity e) {
        entities.add(e);
    }

    @Override
    public void addPlatform(Platform p) {
        platforms.add(p);
        addEntity(p);
    }

    @Override
    public void addEnemy(Enemy e) {
        enemies.add(e);
        addEntity(e);
    }

    @Override
    public void addBackground(BackgroundEntity b) {
        bg.add(b);
        addEntity(b);
    }

    @Override
    public void addFinishLine(double d) {
        finishLine = d;
    }

    @Override
    public Level build() {
        return new LevelImpl(hero,finishLine,entities,platforms,enemies,bg);
    }
}
