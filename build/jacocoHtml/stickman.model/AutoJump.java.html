package stickman.model;

public class AutoJump implements AutoStrategy{
    private double posY;
    private boolean isComingDown;
    private boolean isJumping;
    private double upB;
    private double downB;
    private double height;
    private double velY;

    @Override
    public void getInfo(Enemy e) {
        posY = e.getYPos();
        isComingDown = e.isComingDown;
        isJumping = e.isJumping;
        upB = e.upB;
        downB = e.downB;
        height = e.height;
        velY = e.velY;
    }

    @Override
    public double getValue() {
        if(! isJumping && posY + height < downB){
            isComingDown = true;
        }

        if( isJumping){
            if( posY + velY < upB){
                posY = upB;
                isJumping = false;
                isComingDown = true;
            }else{
                posY += velY;
            }
        }else if( isComingDown){
            if ( posY + height > downB){
                posY = downB - height;
                isComingDown = false;
            }else{
                posY -= velY;
            }
        }
        return posY;
    }

    @Override
    public void update(Enemy e){
        if( isJumping != e.isJumping){
            e.isJumping = isJumping;
        }

        if (isComingDown != e.isComingDown){
            e.isComingDown = isComingDown;
        }

    }
}
