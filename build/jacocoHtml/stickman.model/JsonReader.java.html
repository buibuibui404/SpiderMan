package stickman.model;

import org.json.simple.JSONObject;
import org.json.simple.parser.JSONParser;
import org.json.simple.parser.ParseException;

import java.io.FileNotFoundException;
import java.io.FileReader;
import java.io.IOException;

public class JsonReader {
    private JSONParser parser;
    private String path;
    private JSONObject ob;

    public JsonReader(String name){
        // if other type will be used, just add new method here
        this.parser = new JSONParser();
        path = "src/main/java/stickman/" + name;
        try(FileReader reader = new FileReader(path)){
            System.out.println(path);
            Object obj = parser.parse(reader);
            JSONObject ls = (JSONObject) obj;
            this.ob = ls;

        }catch (FileNotFoundException e){
            e.printStackTrace();
        }catch (IOException e){
            e.printStackTrace();
        }catch (ParseException e){
            e.printStackTrace();
        }
    }

    public String getString(String s){
        return (String) this.ob.get(s);

    }

    public int getInt(String s){ return Integer.parseInt( (String) this.ob.get(s)); }

    public double getDouble(String s){
        return (double) this.ob.get(s);
    }

    public double getInSideDouble(String sub, String s){
        return (double)  ( ( (JSONObject) this.ob.get(sub)).get(s) );
    }

    public JSONObject getJSONobject(String aim){
        return (JSONObject) this.ob.get(aim);
    }
}
