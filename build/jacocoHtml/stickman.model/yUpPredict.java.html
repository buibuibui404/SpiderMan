package stickman.model;

public class yUpPredict implements Strategy {
    private double yPos;

    private double mainX;
    private double mainY;
//    private double mainH;
    private double mainW;

    private double distance;

    public yUpPredict() {
        this.yPos = 0;
        this.distance = 800;
    }

    @Override
    public void update(Entity e) {
//        this.mainH = e.getHeight();
        this.mainW = e.getWidth();
        this.mainX = e.getXPos();
        this.mainY = e.getYPos();
    }


    @Override
    public void getInfo(Entity e) {
        if ( this.distance >= this.mainY - ( e.getYPos() + e.getHeight() ) && this.mainY - ( e.getYPos() + e.getHeight() ) >= 0){
            this.distance = this.mainY - ( e.getYPos() + e.getHeight() );
            if ( ! ( e.getXPos() > this.mainX + this.mainW  || e.getXPos() + e.getWidth() < this.mainX )){
                this.yPos = e.getYPos() + e.getHeight();
//                System.out.println(this.yPos);
            }
        } else if ( this.mainY - ( e.getYPos() + e.getHeight()) < 0 && this.mainY - e.getYPos() > 0){
            if ( ! ( e.getXPos() > this.mainX + this.mainW  || e.getXPos() + e.getWidth() < this.mainX )){
                this.yPos = e.getYPos() + e.getHeight();
                this.distance = this.mainY - ( e.getYPos() + e.getHeight() );
            }
        }

    }

    @Override
    public double getBound() {
//        System.out.println(yPos);
        return this.yPos;
    }

    @Override
    public void reset() {
        this.yPos = 0;
        this.distance = 800;
    }
}
