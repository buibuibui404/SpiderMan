package stickman.model;

public class BlueSlime extends Enemy {
    private AutoStrategy autoDown;
    private AutoStrategy autoMove;

    public BlueSlime(double posX, double posY){
        super(posX, posY);
        this.velX = 1;
        this.velY = -1;
        this.mainPath = "slimeBa.png";

        this.width = 42;
        this.height = 30;

        this.Hp = 1;

        this.autoDown = new AutoDown();
        this.autoMove = new AutoXmove();
    }


    @Override
    void changYLocation() {
        autoDown.getInfo(this);
        posY = autoDown.getValue();
    }

    @Override
    void changXLocation() {
        autoMove.getInfo(this);
        this.posX = autoMove.getValue();
        autoMove.update(this);
    }

    @Override
    void tick() {
        changXLocation();
        changYLocation();

        if( this.timer < 60){
            setXDirection(true);
        }else{
            setXDirection(false);
        }

        this.timer += 1;
        if( this.timer >= 120){
            this.timer = 0;
        }

    }
}
