package stickman.model;

public abstract class MovableEntity extends Entity{

    protected int timer;
    protected int jumpTimer;

    protected boolean ismovingRight;
    protected boolean ismovingLeft;
    protected boolean isJumping;
    protected boolean isComingDown;

    protected double velX;
    protected double velY;
    protected boolean Xdirection = true; // ture for right, false for left
    protected boolean Ydirection = true;

    public double setXVelocity(double velX){
        return this.velX = velX;
    }   // set the horizontal velocity

    public double setYVelocity(double velY){
        return this.velY = velY;
    }   // set the vertical velocity

    public boolean setXDirection(boolean d){
        return this.Xdirection = d;
    }      // change the vertical direction  true for right false for left

    public boolean setYDirection(boolean d){
        return this.Ydirection = d; // by far only for hero, may be added in MovableEntity Interface in the next stage
    }
                                        // y direction can be added for further development
    abstract void changYLocation();     // change the vertical location using y velocity
    abstract void changXLocation();     // change the horizontal location using x velocity
    abstract void tick();
}
