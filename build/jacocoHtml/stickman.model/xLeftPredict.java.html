package stickman.model;

public class xLeftPredict implements Strategy {
    private double xPos;

    private double mainX;
    private double mainY;
    private double mainH;
    private double mainW;

    private double LeftPoint;

    private double distance;

    public xLeftPredict() {
        this.xPos = -1;
        this.distance = 100000;

    }

    @Override
    public void update(Entity e) {
        this.mainH = e.getHeight();
        this.mainW = e.getWidth();
        this.mainX = e.getXPos();
        this.mainY = e.getYPos();
        this.LeftPoint = this.mainX;
    }

    @Override
    public void getInfo(Entity e) {
//        System.out.println(this.LeftPoint - ( e.getXPos() + e.getWidth() ));
//        System.out.println(this.LeftPoint);
//        System.out.println(distance);
//        System.out.println();
        if ( this.distance >= this.LeftPoint - ( e.getXPos() + e.getWidth() ) && this.LeftPoint - ( e.getXPos() + e.getWidth() ) >= 0){
            this.distance = this.LeftPoint - ( e.getXPos() + e.getWidth());
//            System.out.println(e);
//            System.out.println(e.getXPos());
//            System.out.println(e.getYPos());
//            System.out.println( xPos);
//            System.out.println( ! ( e.getYPos() > this.mainH + this.mainY || e.getYPos() + e.getHeight() < this.mainY ));
            if ( ! ( e.getYPos() >= this.mainH + this.mainY || e.getYPos() + e.getHeight() <= this.mainY ) ){
                this.xPos = e.getXPos() + e.getWidth();
//                System.out.println( xPos);
            }
        } else if ( this.LeftPoint - ( e.getXPos() + e.getWidth() ) < 0 && this.LeftPoint > e.getXPos() ){
            this.distance = this.LeftPoint - ( e.getXPos() + e.getWidth());
            if ( ! ( e.getYPos() >= this.mainH + this.mainY || e.getYPos() + e.getHeight() <= this.mainY ) ){
                this.xPos = e.getXPos() + e.getWidth();
//                System.out.println( xPos);
            }
        }

    }

    @Override
    public double getBound() {
        return this.xPos;
    }

    @Override
    public void reset() {
        this.xPos = -1;
        this.distance = 100000;
    }
}
