package stickman.model;

public class AutoXmove implements AutoStrategy{
    private boolean Xdirection;
    private double posX;
    private double velX;
    private double rightB;
    private double leftB;

    @Override
    public void getInfo(Enemy e) {
        Xdirection = e.Xdirection;
        posX = e.posX;
        velX = e.velX;
        rightB = e.rightB;
        leftB = e.leftB;

    }

    @Override
    public double getValue() {
        if(Xdirection){
            if ( this.posX + this.velX > this.rightB){
                Xdirection = false;
            }else{
                this.posX += this.velX;
            }
        }else {
            if ( this.posX - this.velX < this.leftB){
                Xdirection = true;
            }else{
                this.posX -= this.velX;
            }
        }
        return posX;
    }

    @Override
    public void update(Enemy e) {
        if (Xdirection != e.Xdirection){
            e.Xdirection = Xdirection;
        }
    }
}
