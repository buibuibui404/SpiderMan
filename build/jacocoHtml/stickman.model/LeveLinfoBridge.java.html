package stickman.model;

import org.json.simple.JSONArray;
import org.json.simple.JSONObject;

import java.util.LinkedList;
import java.util.List;


public class LeveLinfoBridge {
    static List<List<Double>>  PlatformPositions;
    static List<List<Double>>  BlueSlimePositions;
    static List<List<Double>>  GreenSlimePosition;
    static List<List<Double>>  RedSlimePosition;
    static List<List<Double>>  CloudsPosition;

    static double cloudVelocity;
    static String stickmanSize;
    static double stickmanPosX;
    static double finishLine;
    static int stickmanLife;

    private static JsonReader reader;

    public static void hardcodeRead(String path){
        reader = new JsonReader(path);
        cloudVelocity = reader.getDouble("cloudVelocity");
        stickmanSize = reader.getString("stickmanSize");
        stickmanPosX = reader.getInSideDouble("stickmanPos", "x");
        stickmanLife = reader.getInt("stickmanLife");
        finishLine = reader.getDouble("finish line");
        stickmanLife = reader.getInt("stickmanLife");

        PlatformPositions = radar("Platform Position", reader.getInt("Platform Amount"));
        BlueSlimePositions = radar("Blue Slime Position", reader.getInt("The amount of blue slime"));
        GreenSlimePosition = radar("Green Slime Position", reader.getInt("The amount of green slime"));
        RedSlimePosition = radar("Red Slime Position", reader.getInt("The amount of red slime"));
        CloudsPosition = radar("Cloud Position", reader.getInt("The amount of cloud"));

    }


    private  static List<List<Double>> radar(String aim, int num){
        List<List<Double>> result = new LinkedList<>();
        JSONObject obj = reader.getJSONobject(aim);
        for(Integer i = 1; i <= num; i++){
            JSONArray ls = (JSONArray) obj.get(i.toString());
            List<Double> tempo = new LinkedList<>();
            tempo.add((Double) ls.get(0));
            tempo.add((Double) ls.get(1));
            result.add(tempo);
        }
        return result;
    }







}
