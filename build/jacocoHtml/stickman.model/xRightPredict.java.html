package stickman.model;

public class xRightPredict implements Strategy{
    private double xPos;

    private double mainX;
    private double mainY;
    private double mainH;
    private double mainW;

    private double rightPoint;

    private double distance;

    public xRightPredict(){
        this.xPos = 100000;
        this.distance = 100000;

    }

    @Override
    public void update(Entity e) {
        this.mainH = e.getHeight();
        this.mainW = e.getWidth();
        this.mainX = e.getXPos();
        this.mainY = e.getYPos();
        this.rightPoint = this.mainX + this.mainW;

    }

    @Override
    public void getInfo(Entity e) {
//        System.out.println(e.getYPos());
//        System.out.println(e.getYPos() + e.getHeight());
//        System.out.println(mainH + mainY);
//        System.out.println(mainY);
        if ( this.distance >= e.getXPos() - this.rightPoint && e.getXPos() - this.rightPoint >= 0){
            this.distance = e.getXPos() - this.rightPoint;
//            System.out.println( xPos);
//            System.out.println( ! ( e.getYPos() > this.mainH + this.mainY || e.getYPos() + e.getHeight() < this.mainY ));
            if ( ! ( e.getYPos() >= this.mainH + this.mainY || e.getYPos() + e.getHeight() <= this.mainY ) ){
                this.xPos = e.getXPos();
            }
        } else if ( e.getXPos() - this.rightPoint < 0 && e.getXPos() > this.mainX){
            this.distance = e.getXPos() - this.rightPoint;
            if ( ! ( e.getYPos() >= this.mainH + this.mainY || e.getYPos() + e.getHeight() <= this.mainY ) ){
                this.xPos = e.getXPos();
            }
        }
    }

    @Override
    public void reset(){
        this.xPos = 100000;
        this.distance = 100000;
    }


    @Override
    public double getBound() {
        return this.xPos;
    }
}
