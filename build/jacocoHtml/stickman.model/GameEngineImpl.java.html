package stickman.model;

public class GameEngineImpl implements GameEngine {
    private Level lv;

    public GameEngineImpl(String info) {
        LeveLinfoBridge.hardcodeRead(info);
        LevelBuilder lb = new LevelBuilderImpl();

        lb.addFinishLine(LeveLinfoBridge.finishLine);
        lb.setHero(new Hero(LeveLinfoBridge.stickmanSize, LeveLinfoBridge.stickmanPosX, LeveLinfoBridge.stickmanLife));

        for(int i= 0; i < LeveLinfoBridge.PlatformPositions.size(); i++){
            lb.addPlatform(new NormalPlatform(LeveLinfoBridge.PlatformPositions.get(i).get(0),
                    LeveLinfoBridge.PlatformPositions.get(i).get(1)
                    ));
        }

        for(int i = 0; i < LeveLinfoBridge.RedSlimePosition.size(); i++){
            lb.addEnemy(new RedSlime(LeveLinfoBridge.RedSlimePosition.get(i).get(0),
                    LeveLinfoBridge.RedSlimePosition.get(i).get(1)
            ));
        }

        for(int i = 0; i < LeveLinfoBridge.GreenSlimePosition.size(); i++){
            lb.addEnemy(new GreenSlime(LeveLinfoBridge.GreenSlimePosition.get(i).get(0),
                    LeveLinfoBridge.GreenSlimePosition.get(i).get(1)
            ));
        }

        for(int i = 0; i < LeveLinfoBridge.BlueSlimePositions.size(); i++){
            lb.addEnemy(new BlueSlime(LeveLinfoBridge.BlueSlimePositions.get(i).get(0),
                    LeveLinfoBridge.BlueSlimePositions.get(i).get(1)
            ));
        }

        for(int i = 0; i < LeveLinfoBridge.CloudsPosition.size(); i++){
            lb.addBackground(new Cloud(LeveLinfoBridge.CloudsPosition.get(i).get(0),
                    LeveLinfoBridge.CloudsPosition.get(i).get(1),
                    LeveLinfoBridge.cloudVelocity
            ));
        }

        lb.addBackground(new FinishLine(LeveLinfoBridge.finishLine,330,0));



        lv = lb.build();

    }

    @Override
    public Level getCurrentLevel() {
        return this.lv;
    }

    @Override
    public void startLevel() {
    }

    @Override
    public boolean jump() {
        this.lv.jump();
        return true;
    }

    @Override
    public boolean moveLeft() {
        this.lv.moveLeft();
        return true;
    }

    @Override
    public boolean moveRight() {
        this.lv.moveRight();
        return true;
    }

    @Override
    public boolean stopMoving() {
        this.lv.stopMoving();
        return true;
    }

    @Override
    public void tick() {
        this.lv.tick();
    }
}
