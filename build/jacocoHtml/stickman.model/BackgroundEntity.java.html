package stickman.model;

public abstract class BackgroundEntity extends Entity {
    protected double velX;

    public BackgroundEntity(String path, double XPos, double YPos, double vX, double height, double width){
        this.layer = Layer.BACKGROUND;
        this.mainPath = path;
        this.posX = XPos;
        this.posY = YPos;
        this.velX = vX;
        this.height = height;
        this.width = width;
    }

    public BackgroundEntity(String path, double XPos, double YPos, double vX){
        this.layer = Layer.BACKGROUND;
        this.mainPath = path;
        this.posX = XPos;
        this.posY = YPos;
        this.velX = vX;
        this.height = height;
        this.width = width;
    }

    public void changXLocation(){
        this.posX += this.velX / 60;
    }

    public void tick(){
        this.changXLocation();
    }
}
