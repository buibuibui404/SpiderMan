package stickman.view;


import javafx.scene.Node;
import javafx.scene.control.Label;
import javafx.scene.paint.Color;
import stickman.model.Entity;

public class TimeWidget {
    private Integer timeSeconds = 0;
    private Integer timeM = 0;
    private Label timerLabel;
    private Integer timer = 0;


    public TimeWidget(){
        timerLabel = new Label();
        timerLabel.setText(timeM + " Min " +timeSeconds.toString()+ " S");
        timerLabel.setTextFill(Color.BLACK);
        timerLabel.setLayoutY(0);
        timerLabel.setLayoutX(0);

    }

    public void updatePosX(double x){
        timerLabel.setLayoutX(x);
    }

    public Label getLabel(){
        return timerLabel;
    }

    public void addOneSecond(){
        if( timer % 120 == 0 && timer != 0){
//            System.out.println(timer);
//            System.out.println("!!!!"+ timeSeconds);
            timeSeconds += 1;
            if( timeSeconds % 60 == 0 && timeSeconds != 0){
                timeSeconds = 0;
                timeM += 1;
            }
            timerLabel.setText(timeM + " Min "+ timeSeconds.toString() + " S");
            timer = 1;
        }else{
            timer += 1;
        }
    }
}
